 50def ntia_very_high_gain_model_point(theta,gain =50):
 51    """From NTIA / ITS TM 09-461, A statistical gain antenna model that determines the radar antenna gain in the azimuth orientation. Assumes a maximum
 52    gain greater than and 48 dBi"""
 53    assert gain>48, "gain is too low to use this model, ntia_high_gain or ntia_medium_gain"
 54    theta = abs(theta)
 55    theta_m = 50*(.25*gain+7)**(.5)/10**(gain/20)
 56    theta_r = 27.466*10**(-.3*gain/10)
 57    theta_b = 48
 58    gain_out  = 0
 59    if 0<=theta<np.pi/180 *theta_m:
 60        gain_out = gain - 4*10**(-4)*(10**(gain/10))*(theta*180/np.pi)**2
 61    elif np.pi/180 *theta_m<=theta<np.pi/180 *theta_r:
 62        gain_out = .75*gain - 7
 63    elif np.pi/180 *theta_r<=theta<np.pi/180 *theta_b:
 64        gain_out = 29-25*np.log10(theta*180/np.pi)
 65    else:

123def calculate_relative_angle(x1,y1,x2,y2):
124        """Calculates the relative angle between two locations specified as x1,y1 and x2,y2"""
125        mag_a = np.sqrt(x2**2+y2**2)
126        mag_b = np.sqrt(x1**2+y1**2)
127        inverse_cos =  np.arccos((x1*x2+y1*y2)/(mag_a*mag_b))
128        v1 = [x1,y1,0]
129        v2= [x2,y2,0]
130        cross_product =np.cross(v1,v2)
131        if cross_product[2]<0:
132            angle = -inverse_cos
133        else:
134            angle = inverse_cos
135        return angle

141def node_distance(node1,node2):
142    "returns the distance between node1 and node2"
143    return np.sqrt((node1.location[0]-node2.location[0])**2+(node1.location[1]-node2.location[1])**2)

145def node_to_node_power(node1,node2,wavelength = C/3.75e9):
146    """Returns the loss in dB between 2 nodes for a specified wavelength"""
147    distance = node_distance(node1,node2)
148    rx_angle = node1.calculate_relative_angle(node2.location[0]-node1.location[0],node2.location[1]-node1.location[1])
149    tx_angle = node2.calculate_relative_angle(node1.location[0]-node2.location[1],node1.location[1]-node2.location[1])
150    gain_rx = node1.antenna_pattern(rx_angle)
151    gain_tx = node2.antenna_pattern(tx_angle)
152    power_tx = node2.power
153    power_rx = power_tx + gain_rx+ gain_tx + 20*np.log10(wavelength/(4*np.pi*distance))
154    if isinstance(power_rx,np.ndarray):
155        power_rx = power_rx[0]
156    return power_rx

158def node_to_node_loss(node1,node2,wavelength = C/3.75e9):
159    """Returns the loss in dB between 2 nodes"""
160    distance = node_distance(node1,node2)
161    rx_angle = node1.calculate_relative_angle(node2.location[0]-node1.location[0],node2.location[1]-node1.location[1])
162    tx_angle = node2.calculate_relative_angle(node1.location[0]-node2.location[1],node1.location[1]-node2.location[1])
163    gain_rx = node1.antenna_pattern(rx_angle)
164    gain_tx = node2.antenna_pattern(tx_angle)
165    loss = gain_rx+ gain_tx + 20*np.log10(wavelength/(4*np.pi*distance))
166    return loss

235class Node():
236    """A simple model of a receiver or transmitter that has location, direction, antenna pattern and an optional id.
237      Has a convenience option for calculating the relative angle to any point, and also an azimuthal plot of the antenna function"""
238    def __init__(self,location,direction=[0,1],antenna_pattern=simple_directional_gain,id=None):
239        self.direction = direction
240        self.location = location
241        self.antenna_pattern = antenna_pattern
242        self.id = id
243
244    def calculate_relative_angle(self,x,y):
245        return vcal(self.direction[0],self.direction[1],x,y)
246    
247    def plot_antenna_pattern(self,**options):
248        fig, ax = plt.subplots(subplot_kw={'projection': 'polar'})
249        theta = np.linspace(-np.pi, np.pi, 100)
250        relative_theta = self.calculate_relative_angle(np.cos(theta),np.sin(theta))
251        r = self.antenna_pattern(relative_theta)
252        theta_direction = np.arctan2(self.direction[1],self.direction[0])
253        ax.plot(theta_direction+relative_theta,r,color="b",marker="D",linestyle="solid")
254        ax.set_title(f"Antenna Pattern Gain", va='bottom')
255        ax.grid(True)
